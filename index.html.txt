    <script>
        const myPhoneNumber = "6282322818690"; 

        const codeProducts = [
            { name: "VIP 7D", price: "21.000" },
            { name: "VIP 30D", price: "63.000" },
            { name: "KVIP 7D", price: "38.000" },
            { name: "KVIP 30D", price: "105.000" },
            { name: "SVIP 7D", price: "47.000" },
            { name: "SVIP 30D", price: "136.000" }
        ];

        const jokiProducts = [
            { name: "Joki War Normal", price: "12.000", desc: "Proses 10M-3D, Bonus SG (Hoki)" },
            { name: "Joki Prio", price: "20.000", desc: "No Antri + Free Setting RF" },
            { name: "Joki Req Server SG", price: "35.000", desc: "Tingkat kesulitan tinggi" }
        ];

        function renderProducts() {
            const codeContainer = document.getElementById('list-code');
            const jokiContainer = document.getElementById('list-joki');

            codeProducts.forEach(item => {
                const div = document.createElement('div');
                div.className = "bg-white p-4 rounded-xl border-2 border-gray-100 shadow-sm cursor-pointer product-card transition-all flex justify-between items-center";
                div.innerHTML = `<div><p class="font-bold text-gray-800">${item.name}</p><p class="text-red-600 font-bold">Rp ${item.price}</p></div><span class="text-xs bg-red-100 text-red-600 px-2 py-1 rounded-full font-bold">Pilih</span>`;
                div.onclick = () => selectItem(item.name, item.price, "CODE");
                codeContainer.appendChild(div);
            });

            jokiProducts.forEach(item => {
                const div = document.createElement('div');
                div.className = "bg-white p-4 rounded-xl border-2 border-gray-100 shadow-sm cursor-pointer product-card transition-all";
                div.innerHTML = `<div class="flex justify-between items-start"><div><p class="font-bold text-gray-800">${item.name}</p><p class="text-blue-600 font-bold">Rp ${item.price}</p></div><span class="text-xs bg-blue-100 text-blue-600 px-2 py-1 rounded-full font-bold">Pilih</span></div><p class="text-[10px] text-gray-500 mt-1 italic">${item.desc}</p>`;
                div.onclick = () => selectItem(item.name, item.price, "JOKI");
                jokiContainer.appendChild(div);
            });
        }

        function selectItem(name, price, type) {
            document.getElementById('selected-product').value = `[${type}] ${name} - Rp ${price}`;
        }

        function showQRIS() {
            document.getElementById('qris-modal').style.display = 'flex';
        }

        function hideQRIS() {
            document.getElementById('qris-modal').style.display = 'none';
        }

        function sendToWhatsApp() {
            const name = document.getElementById('buyer-name').value;
            const product = document.getElementById('selected-product').value;

            if (!name || !product) {
                alert("Silakan isi nama dan pilih produk!");
                return;
            }

            const message = `*PESANAN RED FINGER - NEON STORE*\n` +
                            `------------------------------\n` +
                            `*Nama:* ${name}\n` +
                            `*Pesanan:* ${product}\n` +
                            `*Metode:* QRIS (Neon)\n` +
                            `------------------------------\n` +
                            `Saya sudah melakukan pembayaran/akan membayar via QRIS. Mohon diproses min!`;

            const url = `https://wa.me/${myPhoneNumber}?text=${encodeURIComponent(message)}`;
            window.open(url, '_blank');
        }

        renderProducts();
    </script>
